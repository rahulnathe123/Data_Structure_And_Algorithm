8. Create an application for Students Record using AngularJS.




<!DOCTYPE html>
<html ng-app="studentApp">
    <head>
        <title>Student Record</title>
        <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
        <style>
            table, th, td {
                border: 1px solid black;
                border-collapse: collapse;
                padding: 8px;
            }
            th 
            {
                background-color: #f2f2f2;
            }
        </style>
    </head>
    
    <body ng-controller="studentCtrl">

        <h2>Student Record</h2>
  
        <form ng-submit="addStudent()">
            <input type="text" ng-model="newStudent.name" placeholder="Enter Name" required>
            <input type="number" ng-model="newStudent.roll" placeholder="Enter Roll No" required>
            <input type="text" ng-model="newStudent.course" placeholder="Enter Course" required>
            <button type="submit">Add Student</button>
        </form>

        <br>
        <table>
            <tr>
            <th>Roll No</th>
            <th>Name</th>
            <th>Course</th>
            <th>Action</th>
            </tr>
                <tr ng-repeat="student in students">
                <td>{{student.roll}}</td>
                <td>{{student.name}}</td>
                <td>{{student.course}}</td>
                <td>
                    <button ng-click="removeStudent($index)">Delete</button>
                </td>
            </tr>
        </table>

        <script>
            angular.module("studentApp", []).controller("studentCtrl", function($scope) 
            {
                $scope.students = [
                    {roll: 101, name: "Amit", course: "BCA"},
                    {roll: 102, name: "Priya", course: "MCA"},
                    {roll: 103, name: "Rahul", course: "B.Tech"}
                ];

                $scope.addStudent = function() {
                    $scope.students.push({
                        roll: $scope.newStudent.roll,
                        name: $scope.newStudent.name,
                        course: $scope.newStudent.course
                    });
                    $scope.newStudent = {};
                };

                $scope.removeStudent = function(index) {
                    $scope.students.splice(index, 1);
                };
            });
        </script>
    </body>
</html>
