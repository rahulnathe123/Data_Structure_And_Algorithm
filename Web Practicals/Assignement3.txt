    <script>
        function validateForm() 
        {       
            var pid = document.getElementById("pid").value;
            var name = document.getElementById("name").value;
            var address = document.getElementById("address").value;
            var city = document.getElementById("city").value;
            var contact = document.getElementById("contact").value;
            var dob = document.getElementById("dob").value;
            if (pid === "") 
            {           
                alert("Please enter Patient ID.");
                return false;  
            } 
 
            if (name === "")
            {         
                alert("Please enter Patient Name.");         
                return false; 
            } 
            if (!/^[A-Za-z\s]+$/.test(name)) 
            {         
                alert("Name should contain alphabets only.");     
                return false; 
            } 
 
            if (address === "") 
            {
                alert("Please enter Address.");         
                return false; 
            } 
 
            if (city === "")
            {         
                alert("Please enter City.");         
                return false; 
            } 
            if (!/^[A-Za-z\s]+$/.test(city))
            {         
                alert("City should contain alphabets only.");         
                return false; 
            } 
 
            // Contact validation with separate checks       
            // if (contact === "") 
            // {         
                    //alert("Please enter Contact Number.");         
                    // return false;
                //} 
         
            if (isNaN(contact))
            {         
                alert("Contact Number must contain digits only.");         
                return false; 
            } 
 
            if(contact.length !== 10)
            {         
                alert("Contact Number must be exactly 10 digits.");         
                return false; 
            } 
            if(contact.charAt(0) === "0") 
            {         
                alert("Contact Number should not start with 0.");         return false; 
            } 
 
            if (dob === "")
            {         
                alert("Please enter Date of Birth.");
                return false; 
            } 
            alert("Form submitted successfully!");       
            return true;     
        }

    </script>
